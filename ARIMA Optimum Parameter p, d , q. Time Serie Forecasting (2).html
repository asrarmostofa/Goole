import itertools
from statsmodels.tsa.arima.model import ARIMA
from sklearn.metrics import mean_squared_error
import numpy as np

p = range(0, 8)
q = range(0, 8)
d = range(0, 2)

pdq_combination = list(itertools.product(p, d, q))
rmse = []
order1 = []

output_file = 'C:/Users/Asrar/Downloads/arima_results.txt'  # Change this path accordingly

with open(output_file, 'w') as f:
    for pdq in pdq_combination:
        try:
            model = ARIMA(train, order=pdq).fit()
            predicted_values = model.predict(start=len(train), end=(len(ts) - 1))

            # Assuming your 'test' variable has the actual values
            # and 'predicted_values' has the predicted values
            test = test[:len(predicted_values)]

            # Calculate the root mean squared error
            error = np.sqrt(mean_squared_error(test, predicted_values))
            rmse.append(error)
            order1.append(pdq)

            result_line = f"ARIMA{pdq} - RMSE: {error}\n"
            print(result_line)
            f.write(result_line)
        except Exception as e:
            error_message = f"ARIMA{pdq} - Error: {str(e)}\n"
            print(error_message)
            f.write(error_message)

# Find the best model with the minimum RMSE
best_rmse_index = np.argmin(rmse)
best_order = order1[best_rmse_index]

best_result = f"Best Model - ARIMA{best_order} with RMSE: {rmse[best_rmse_index]}\n"
print(best_result)
with open(output_file, 'a') as f:
    f.write(best_result)
